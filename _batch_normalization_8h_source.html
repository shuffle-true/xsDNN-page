92  void forward(const Matrix &prev_layer_data) override {
93  const long ncols = prev_layer_data.cols();
94  m_z.resize(this->m_in_size, ncols);
95  m_a.resize(this->m_in_size, ncols);
96 
97  Vector mean = (workflow == "train") ? mean_curr : m_mean;
98  Vector var = (workflow == "train") ? var_curr : m_var;
99 
100  if (workflow == "train") {
101  internal::math::update_statistics(prev_layer_data, this->m_in_size, mean, var);
102  }
103 
104  m_stddev = (var.array() + eps).sqrt();
105 
106  int index = 0;
107  for (auto col: prev_layer_data.colwise()) {
108  m_z.col(index++) = (col.array() - mean.array()) / m_stddev.array();
109  }
110 
111  if (affine_) {
112  m_z = m_z.array().colwise() * m_gammas.array();
113  m_z = m_z.array().colwise() + m_betas.array();
114  }
115 
116  Activation::activate(m_z, m_a);
117  }

130  void backprop(const Matrix &prev_layer_data,
131  const Matrix &next_layer_backprop_data) override {
132  const int ncols = m_z.cols();
133  m_din.resize(this->m_in_size, ncols);
134 
135  Matrix dLz;
136  Activation::apply_jacobian(m_z, m_a, next_layer_backprop_data, dLz);
137 
138  // if Y = (X-mean(X))/(sqrt(var(X)+eps)), then
139  //
140  // dE(L)/dX =
141  // (dL/dz - mean(dL/dz) - mean(dL/dz \cdot Y) \cdot Y)
142  // ./ sqrt(var(X) + eps)
143  //
144 
145  Matrix dLz_dot_y;
146  dLz_dot_y.resize(this->m_in_size, ncols);
147  dLz_dot_y = dLz.cwiseProduct(m_z);
148 
149  Vector mean_dLz, mean_dLz_dot_y;
150  mean_dLz.resize(this->m_in_size);
151  mean_dLz_dot_y.resize(this->m_in_size);
152  mean_dLz = dLz.rowwise().mean();
153  mean_dLz_dot_y = dLz_dot_y.rowwise().mean();
154 
155  int index = 0;
156  for (auto col: m_din.colwise()) {
157  col = dLz.col(index) - mean_dLz - mean_dLz_dot_y.cwiseProduct(m_z.col(index));
158  index += 1;
159  }
160  index = 0;
161  for (auto row: m_din.rowwise()) {
162  row.array() /= m_stddev[index++];
163  }
164 
165  if (affine_) {
166  m_dg = mean_dLz_dot_y;
167  m_db = mean_dLz;
168  }
169  }

177  void update(Optimizer &opt) override {
178  m_mean = moment * m_mean + (1 - moment) * mean_curr;
179  m_var = moment * m_var + (1 - moment) * var_curr;
180 
181  if (affine_) {
182  AlignedMapVec dg(m_dg.data(), m_dg.size());
183  AlignedMapVec g(m_gammas.data(), m_gammas.size());
184  AlignedMapVec db(m_db.data(), m_db.size());
185  AlignedMapVec b(m_betas.data(), m_betas.size());
186 
187  opt.update(dg, g);
188  opt.update(db, b);
189  }
190  }
