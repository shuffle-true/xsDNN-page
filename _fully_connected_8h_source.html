53  m_dw.resize(this->m_in_size, this->m_out_size);

65  void forward(const Matrix &prev_layer_data) override {
66  const long ncols = prev_layer_data.cols();
67 
68  m_z.resize(this->m_out_size, ncols);
69  m_z.noalias() = m_weight.transpose() * prev_layer_data;
70  if (BIAS_ACTIVATE) { m_z.colwise() += m_bias; }
71  m_a.resize(this->m_out_size, ncols);
72  Activation::activate(m_z, m_a);
73  }
74 
77  const Matrix &output() const override { return m_a; }
78 
87  void backprop(const Matrix &prev_layer_data,
88  const Matrix &next_layer_backprop_data) override {
89  const long ncols = prev_layer_data.cols();
90 
91  Matrix &dLz = m_z;
92  Activation::apply_jacobian(m_z, m_a, next_layer_backprop_data, dLz);
93  m_dw.noalias() = prev_layer_data * dLz.transpose() / ncols;
94  if (BIAS_ACTIVATE) { m_db.noalias() = dLz.rowwise().mean(); }
95  m_din.resize(this->m_in_size, ncols);
96  m_din.noalias() = m_weight * dLz;
97  }

101  const Matrix &backprop_data() const override { return m_din; }

106  void update(Optimizer &opt) override {
107  AlignedMapVec dw(m_dw.data(), m_dw.size());
108  AlignedMapVec w(m_weight.data(), m_weight.size());
109  AlignedMapVec db(m_db.data(), m_db.size());
110  AlignedMapVec b(m_bias.data(), m_bias.size());
111 
112  opt.update(dw, w);
113  if (BIAS_ACTIVATE) { opt.update(db, b); }
114 
115  }

119  std::vector<Scalar> get_parametrs() const override {
120  if (BIAS_ACTIVATE) {
121  std::vector<Scalar> res(m_weight.size() + m_bias.size()); // указали кол-во ячеек в этом векторе
122  // просто копируем в этот массив все содержимое
123  std::copy(m_weight.data(), m_weight.data() + m_weight.size(),
124  res.begin()); // все аргументы передаются в виде указателей. Откуда начинаем, где заканчиваем, куда начинаем ложить.
125  std::copy(m_bias.data(), m_bias.data() + m_bias.size(), res.begin() + m_weight.size());
126  return res;
127  } else {
128  std::vector<Scalar> res(m_weight.size());
129  std::copy(m_weight.data(), m_weight.data() + m_weight.size(), res.begin());
130  return res;

145  void set_parametrs(const std::vector<Scalar> &param) override {
146  if (BIAS_ACTIVATE) {
147  // сделаем проверку на равенство длин массивов
148  // static_cast<int> - приведение длины массива к интовому типу данных
149  if (static_cast<int>(param.size()) != (m_weight.size() + m_bias.size())) {
150  throw internal::except::xs_error(
151  "[class FullyConnected]: Parameter size does not match. Check parameter size!");
152  }
153 
154  // если размеры сходятся, то копируем переданные значения в наши массивы
155  std::copy(param.begin(), param.begin() + m_weight.size(), m_weight.data());
156  std::copy(param.begin() + m_weight.size(), param.end(), m_bias.data());
157  } else {
158  if (static_cast<int>(param.size()) != (m_weight.size())) {
159  throw internal::except::xs_error(
160  "[class FullyConnected]: Parameter size does not match. Check parameter size!");
161  }
162 
163  // если размеры сходятся, то копируем переданные значения в наши массивы
164  std::copy(param.begin(), param.begin() + m_weight.size(), m_weight.data());
165  }
