142  void Backprop(const Matrix& input, const Matrix& target)
143  {
144  const unsigned long nlayer = count_layers();
145 
146  if (nlayer <= 0) { return; }
147 
148  // Создадим указатель на первый и последний (скрытый, но не выходной)
149  // слой сетки, это поможет в дальнейшем
150 
151  Layer* first_layer = m_layers[0];
152  Layer* last_layer = m_layers[nlayer - 1];
153 
154  // Начнем распространение с конца сетки
155  m_output->check_target_data(target);
156  m_output->evaluate(last_layer->output(), target);
157 
158  // Если скрытый слой всего один, то 'prev_layer_data' будут выходными данными
159 
160  if (nlayer == 1)
161  {
162  first_layer->backprop(input, last_layer->backprop_data());
163  return;
164  }
165 
166  // Если это условие не выполнено, то вычисляем градиент для последнего скрытого слоя
167  last_layer->backprop(m_layers[nlayer - 2]->output(), m_output->backprop_data());
168 
169  // Теперь пробегаемся по всем слоям и вычисляем градиенты
170 
171  for (unsigned long i = nlayer - 2; i > 0; --i)
172  {
173  m_layers[i]->backprop(m_layers[i - 1]->output(),
174  m_layers[i + 1]->backprop_data());
175  }
176 
177  // Теперь вычисляем грады для нулевого - входного слоя сетки
178 
179  first_layer->backprop(input, m_layers[1]->backprop_data());
180 
181  // На этом backprop окончен
182  }
