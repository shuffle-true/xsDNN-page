115  void backprop(const Matrix& input, const TargetType& target)
116  {
117  const unsigned long nlayer = count_layers();
118 
119  if (nlayer <= 0) { return; }
120 
121  // Создадим указатель на первый и последний (скрытый, но не выходной)
122  // слой сетки, это поможет в дальнейшем
123 
124  Layer* first_layer = m_layers[0];
125  Layer* last_layer = m_layers[nlayer - 1];
126 
127  // Начнем распространение с конца сетки
128  m_output->check_target_data(target);
129  m_output->evaluate(last_layer->output(), target);
130 
131  // Если скрытый слой всего один, то 'prev_layer_data' будут выходными данными
132 
133  if (nlayer == 1)
134  {
135  first_layer->backprop(input, last_layer->backprop_data());
136  return;
137  }
138 
139  // Если это условие не выполнено, то вычисляем градиент для последнего скрытого слоя
140  last_layer->backprop(m_layers[nlayer - 2]->output(), m_output->backprop_data());
141 
142  // Теперь пробегаемся по всем слоям и вычисляем градиенты
143 
144  for (unsigned long i = nlayer - 2; i > 0; --i)
145  {
146  m_layers[i]->backprop(m_layers[i - 1]->output(),
147  m_layers[i + 1]->backprop_data());
148  }
149 
150  // Теперь вычисляем грады для нулевого - входного слоя сетки
151 
152  first_layer->backprop(input, m_layers[1]->backprop_data());
153 
154  // На этом backprop окончен
155  }
