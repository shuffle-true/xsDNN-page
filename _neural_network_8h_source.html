146  void Backprop(const Matrix& input, const Matrix& target)
147  {
148  const unsigned long nlayer = count_layers();
149 
150  if (nlayer <= 0) { return; }
151 
152  // Создадим указатель на первый и последний (скрытый, но не выходной)
153  // слой сетки, это поможет в дальнейшем
154 
155  Layer* first_layer = m_layers[0];
156  Layer* last_layer = m_layers[nlayer - 1];
157 
158  // Начнем распространение с конца сетки
159  m_output->check_target_data(target);
160  m_output->evaluate(last_layer->output(), target);
161 
162  // Если скрытый слой всего один, то 'prev_layer_data' будут выходными данными
163 
164  if (nlayer == 1)
165  {
166  first_layer->backprop(input,
167  last_layer->backprop_data());
168  return;
169  }
170 
171  // Если это условие не выполнено, то вычисляем градиент для последнего скрытого слоя
172  last_layer->backprop(m_layers[nlayer - 2]->output(),
173  m_output->backprop_data());
174 
175  // Теперь пробегаемся по всем слоям и вычисляем градиенты
176 
177  for (unsigned long i = nlayer - 2; i > 0; --i)
178  {
179  m_layers[i]->backprop(m_layers[i - 1]->output(),
180  m_layers[i + 1]->backprop_data());
181  }
182

184 
185  first_layer->backprop(input,
