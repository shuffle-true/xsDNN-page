157  void backprop(const Matrix& input, const TargetType& target)
158  {
159  const unsigned long nlayer = count_layers();
160 
161  if (nlayer <= 0) { return; }
162 
163  // Создадим указатель на первый и последний (скрытый, но не выходной)
164  // слой сетки, это поможет в дальнейшем
165 
166  Layer* first_layer = m_layers[0];
167  Layer* last_layer = m_layers[nlayer - 1];
168 
169  // Начнем распространение с конца сетки
170  m_output->check_target_data(target);
171  m_output->evaluate(last_layer->output(), target);
172 
173  // Если скрытый слой всего один, то 'prev_layer_data' будут выходными данными
174 
175  if (nlayer == 1)
176  {
177  first_layer->backprop(input, last_layer->backprop_data());
178  return;
179  }
180 
181  // Если это условие не выполнено, то вычисляем градиент для последнего скрытого слоя
182  last_layer->backprop(m_layers[nlayer - 2]->output(), m_output->backprop_data());
183 
184  // Теперь пробегаемся по всем слоям и вычисляем градиенты
185 
186  for (unsigned long i = nlayer - 2; i > 0; --i)
187  {
188  m_layers[i]->backprop(m_layers[i - 1]->output(),
189  m_layers[i + 1]->backprop_data());
190  }
191 
192  // Теперь вычисляем грады для нулевого - входного слоя сетки
193 
194  first_layer->backprop(input, m_layers[1]->backprop_data());
195 
196  // На этом backprop окончен
197  }
