14 
15  if ((target_data.cols() != net_output.cols()) || (target_data.rows() != net_output.rows()))

37  void R_calculate(Matrix& target_data, Matrix& net_output)

39

41  {
42  throw std::invalid_argument("[void R_calculate]: Input data have incorrect dimension");
43  }
44 
45  Scalar MSE = (target_data - net_output).squaredNorm();
46  Matrix target_data_mean = target_data.rowwise().mean();
47 
48  for (int i = 0; i < target_data_mean.size(); i++)

86  Matrix static build_confusion_matrix(const Scalar* real, const Scalar* predict, const long& sample_size)
87  {
88  int tp = 0;
89  int tn = 0;
90  int fp = 0;
91  int fn = 0;
92 
93  for (int i = 0; i < sample_size; i++)
94  {
95  if ((real[i] == 0) && (predict[i] == 0)) tn++;
96  else if ((real[i] == 1) && (predict[i] == 1)) tp++;
97  else if ((real[i] == 0) && (predict[i] == 1)) fp++;
98  else if ((real[i] == 1) && (predict[i] == 0)) fn++;
99  }
100 
101  // after calculate value we can build confusion matrix
102  Matrix CM(2,2);
103  CM << tn, fn,
104  fp, tp;
105  return CM;
106  }

113  void accuracy_precision_recall_f1_calculate(const Matrix& y_true, const Matrix& y_pred)
114  {
115 
116  if ((y_true.cols() != y_pred.cols()) || (y_true.rows() != y_pred.rows()))
117  {
118  throw std::invalid_argument("[void accuracy_precision_recall_f1_calculate]: Input data have incorrect dimension");
119  }
120 
121  const long sample_size = y_true.size();
122  const Scalar* real = y_true.data();
123  const Scalar* predict = y_pred.data();
124  Matrix CM = internal::build_confusion_matrix(real, predict, sample_size);
125 
126  Scalar accuracy = CM.diagonal().sum() / CM.sum();
127  Scalar precision = ((CM.diagonal().array()) / (CM.rowwise().sum().array())).mean();
128  Scalar recall = ((CM.diagonal().array().transpose()) / (CM.colwise().sum().array())).mean();
129  Scalar f1_score = 2 * (precision * recall) / (precision + recall);
130 
131  std::cout << "accuracy = " << accuracy << std::endl;
132  std::cout << "precision = " << precision << std::endl;
133  std::cout << "recall = " << recall << std::endl;
134  std::cout << "f1_score = " << f1_score << std::endl;
135  }
