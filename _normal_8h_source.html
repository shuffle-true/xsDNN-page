37  static Scalar NormalZiggurat(RNG& rng) {
38  int iter = 0;
39  do {
40  Scalar B = rng.rand();
41  int stairId = static_cast<int>(B) & 255;
42  Scalar x = Uniform(0, stairWidthNormal[stairId], rng);
43  if (x < stairWidthNormal[stairId + 1])
44  return ((signed)B > 0) ? x : -x;
45  if (stairId == 0)
46  {
47  static Scalar z = -1;
48  Scalar y;
49  if (z > 0)
50  {
51  x = internal::get_exponential(x1, rng);
52  z -= 0.5 * x * x;
53  }
54  if (z <= 0)
55  {
56  do {
57  x = internal::get_exponential(x1, rng);
58  y = internal::get_exponential(1, rng);
59  z = y - 0.5 * x * x;
60  } while (z <= 0);
61  }
62  x += x1;
63  return ((signed)B > 0) ? x : -x;
64  }
65 
66  if (Uniform(stairHeightNormal[stairId - 1], stairHeightNormal[stairId], rng) < std::exp(-.5 * x * x))
67  return ((signed)B > 0) ? x : -x;
68  } while (++iter <= 1e9);
69  return NAN;
70  }
