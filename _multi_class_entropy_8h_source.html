69  Scalar loss() const override {
70  // L = -sum(log(phat) * y)
71  // in = phat
72  // d(L) / d(in) = -y / phat
73  // m_din contains 0 if y = 0, and -1/phat if y = 1
74  Scalar res = Scalar(0);
75  const int nelem = m_din.size();
76  const Scalar *din_data = m_din.data();
77 
78  for (int i = 0; i < nelem; i++) {
79  if (din_data[i] < Scalar(0)) {
80  res += std::log(-din_data[i]);
81  }
82  }
83 
84  return res / m_din.cols();
85  }
