28  inline void parse_mnist_header(std::ifstream& ifs, mnist_header& header)
29  {
30  ifs.read(reinterpret_cast<char *>(&header.magic_number), 4);
31  ifs.read(reinterpret_cast<char *>(&header.num_items), 4);
32  ifs.read(reinterpret_cast<char *>(&header.num_rows), 4);
33  ifs.read(reinterpret_cast<char *>(&header.num_cols), 4);
34 
35  if (is_little_endian()) {
36  reverse_endian(&header.magic_number);
37  reverse_endian(&header.num_items);
38  reverse_endian(&header.num_rows);
39  reverse_endian(&header.num_cols);
40  }
41

47  inline void parse_mnist_image(std::ifstream& ifs,
48  const mnist_header& header,
49  const int width,

51  const Scalar& scale_min,
52  const Scalar& scale_max,
53  const int x_padding,
54  const int y_padding,
55  Eigen::VectorXd& col)

57  typedef Eigen::Matrix<uint8_t, 1, Eigen::Dynamic> Vector;
58 
59  const int w = header.num_cols + 2 * x_padding;
60  const int h = header.num_rows + 2 * y_padding;
61 
62  Vector image_vec(header.num_rows * header.num_cols);
63 
64  ifs.read(reinterpret_cast<char *>(image_vec.data()), header.num_rows * header.num_cols);
65 
66  col.resize(w * h);
67  Scalar* arr = col.data();
68  unsigned char* image_vec_arr = image_vec.data();
69 
70  for (uint32_t y = 0; y < header.num_rows; y++)
71  for (uint32_t x = 0; x < header.num_cols; x++)
72  {
73  arr[w * (y + y_padding) + x + x_padding] =
74  (image_vec_arr[y * header.num_cols + x] / Scalar (255)) *
75  (scale_max - scale_min) +
76  scale_min;

87  inline void parse_mnist_label(const std::string& label_filename, Matrix& label)
88  {
89  std::ifstream ifs(label_filename.c_str(), std::ios::in | std::ios::binary);
90 
91  if (ifs.bad() || ifs.fail())
92  {
93  throw std::invalid_argument("[inline void parse_mnist_label] Error while opening file");
94  }
95 
96  uint32_t magic_number, num_items;
97 
98  ifs.read(reinterpret_cast<char *>(&magic_number), 4);
99  ifs.read(reinterpret_cast<char *>(&num_items), 4);
100 
101  if (internal::is_little_endian()) {
102  internal::reverse_endian(&magic_number);
103  internal::reverse_endian(&num_items);
104  }
105 
106  if (magic_number != 0x00000801 || num_items <= 0)
107  {
108  throw std::invalid_argument("[inline void parse_mnist_label] MNIST label-file format unknown");
109  }
110 
111  label.resize(10, num_items);
112  label.setZero();
113 
114  for (uint32_t i = 0; i < num_items; i++)
115  {
116  uint8_t label_;
117  Scalar* arr = label.col(i).data();
118 
119  ifs.read(reinterpret_cast<char *>(&label_), 1);
120 
121  const uint8_t index = static_cast<uint8_t>(label_);
122 
123  arr[index] = 1;
124  }
125  }
