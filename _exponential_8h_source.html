23  static void setupExpTables() {
24  stairHeight[0] = std::exp(-x1);
25  stairWidth[0] = A / stairHeight[0];
26  stairWidth[256] = 0;
27  for (unsigned i = 1; i <= 255; ++i)
28  {
29  stairWidth[i] = -std::log(stairHeight[i - 1]);
30  stairHeight[i] = stairHeight[i - 1] + A / stairWidth[i];
31  }
32  }

34  static Scalar ExpZiggurat(RNG& rng) {
35  int iter = 0;
36  do {
37  int stairId = static_cast<int>(rng.rand()) & 255;
38  Scalar x = Uniform(0, stairWidth[stairId], rng);
39  if (x < stairWidth[stairId + 1])
40  return x;
41  if (stairId == 0)
42  return x1 + ExpZiggurat(rng);
43  if (Uniform(stairHeight[stairId - 1], stairHeight[stairId], rng) < std::exp(-x))
44  return x;
45  } while (++iter <= 1e9);
46  return NAN;
47  }
